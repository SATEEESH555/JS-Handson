Q.2): "Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) Input: abcadeecfb

Ans: 
             
                function countAlphabet(s)
                {
                  var d = new Map();
                 
                  s.split('').forEach(element => {
                     
                        if(d.has(element))
                        {
                            d.set(element, d.get(element)+1);
                        }
                        else
                            d.set(element, 1);
                  });
                 
                  s.split('').forEach(element => {
                    if(d.has(element) && d.get(element)!=0)
                    {
                      console.log( element + "=" + d.get(element) + " ");
                      d.set(element, 0);
                    }
                  });
                }
                countAlphabet("abcadeecfb");
