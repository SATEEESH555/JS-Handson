Q.7): Write a program which takes a number from user and check whether number is prime number or not a prime number.

Ans: 
             
                const Prime_Check = (N) => 
                {
                var count = 0;
                if (N === 0 || N === 1){
                  count = 1;
                }
                for(let i = 2; i<=N/2; i++) {
                    if(N%i === 0){
                        count++;
                    }
                }
                if(count > 0){
                    return ("NO");
                }
                else{
                    return ("YES");
                }
                };
